65 template <typename Precision, typename I>
66 Impl(const MeshGLP<Precision, I>& meshGL) {
67 const uint32_t numVert = meshGL.NumVert();
68 const uint32_t numTri = meshGL.NumTri();
69
70 if (meshGL.numProp < 3) {
71 MarkFailure(Error::MissingPositionProperties);
72 return;
73 }
74
75 if (meshGL.mergeFromVert.size() != meshGL.mergeToVert.size()) {
76 MarkFailure(Error::MergeVectorsDifferentLengths);
77 return;
78 }
79
80 if (!meshGL.runTransform.empty() &&
81 12 * meshGL.runOriginalID.size() != meshGL.runTransform.size()) {
82 MarkFailure(Error::TransformWrongLength);
83 return;
84 }
85
86 if (!meshGL.runOriginalID.empty() && !meshGL.runIndex.empty() &&
87 meshGL.runOriginalID.size() + 1 != meshGL.runIndex.size() &&
88 meshGL.runOriginalID.size() != meshGL.runIndex.size()) {
89 MarkFailure(Error::RunIndexWrongLength);
90 return;
91 }
92
93 if (!meshGL.faceID.empty() && meshGL.faceID.size() != meshGL.NumTri()) {
94 MarkFailure(Error::FaceIDWrongLength);
95 return;
96 }
97
98 std::vector<int> prop2vert(numVert);
99 std::iota(prop2vert.begin(), prop2vert.end(), 0);
100 for (size_t i = 0; i < meshGL.mergeFromVert.size(); ++i) {
101 const uint32_t from = meshGL.mergeFromVert[i];
102 const uint32_t to = meshGL.mergeToVert[i];
103 if (from >= numVert || to >= numVert) {
104 MarkFailure(Error::MergeIndexOutOfBounds);
105 return;
106 }
107 prop2vert[from] = to;
108 }
109
110 const auto numProp = meshGL.numProp - 3;
111 meshRelation_.numProp = numProp;
112 meshRelation_.properties.resize(meshGL.NumVert() * numProp);
113 // This will have unreferenced duplicate positions that will be removed by
114 // Impl::RemoveUnreferencedVerts().
115 vertPos_.resize(meshGL.NumVert());
116
117 for (size_t i = 0; i < meshGL.NumVert(); ++i) {
118 for (const int j : {0, 1, 2})
119 vertPos_[i][j] = meshGL.vertProperties[meshGL.numProp * i + j];
120 for (size_t j = 0; j < numProp; ++j)
121 meshRelation_.properties[i * numProp + j] =
122 meshGL.vertProperties[meshGL.numProp * i + 3 + j];
123 }
124
125 halfedgeTangent_.resize(meshGL.halfedgeTangent.size() / 4);
126 for (size_t i = 0; i < halfedgeTangent_.size(); ++i) {
127 for (const int j : {0, 1, 2, 3})
128 halfedgeTangent_[i][j] = meshGL.halfedgeTangent[4 * i + j];
129 }
130
131 Vec<TriRef> triRef;
132 if (!meshGL.runOriginalID.empty()) {
133 auto runIndex = meshGL.runIndex;
134 const auto runEnd = meshGL.triVerts.size();
135 if (runIndex.empty()) {
136 runIndex = {0, static_cast<I>(runEnd)};
137 } else if (runIndex.size() == meshGL.runOriginalID.size()) {
138 runIndex.push_back(runEnd);
139 }
140 triRef.resize(meshGL.NumTri());
141 const auto startID = Impl::ReserveIDs(meshGL.runOriginalID.size());
142 for (size_t i = 0; i < meshGL.runOriginalID.size(); ++i) {
143 const int meshID = startID + i;
144 const int originalID = meshGL.runOriginalID[i];
145 for (size_t tri = runIndex[i] / 3; tri < runIndex[i + 1] / 3; ++tri) {
146 TriRef& ref = triRef[tri];
147 ref.meshID = meshID;
148 ref.originalID = originalID;
149 ref.tri = meshGL.faceID.empty() ? tri : meshGL.faceID[tri];
150 }
151
152 if (meshGL.runTransform.empty()) {
153 meshRelation_.meshIDtransform[meshID] = {originalID};
154 } else {
155 const Precision* m = meshGL.runTransform.data() + 12 * i;
156 meshRelation_.meshIDtransform[meshID] = {
157 originalID,
158 {m[0], m[1], m[2], m[3], m[4], m[5], m[6], m[7], m[8], m[9],
159 m[10], m[11]}};
160 }
161 }
162 }
163
164 Vec<ivec3> triVerts;
165 triVerts.reserve(numTri);
166 for (size_t i = 0; i < numTri; ++i) {
167 ivec3 tri;
168 for (const size_t j : {0, 1, 2}) {
169 uint32_t vert = (uint32_t)meshGL.triVerts[3 * i + j];
170 if (vert >= numVert) {
171 MarkFailure(Error::VertexOutOfBounds);
172 return;
173 }
174 tri[j] = prop2vert[vert];
175 }
176 if (tri[0] != tri[1] && tri[1] != tri[2] && tri[2] != tri[0]) {
177 triVerts.push_back(tri);
178 if (triRef.size() > 0) {
179 meshRelation_.triRef.push_back(triRef[i]);
180 }
181 if (numProp > 0) {
182 meshRelation_.triProperties.push_back(
183 ivec3(static_cast<uint32_t>(meshGL.triVerts[3 * i]),
184 static_cast<uint32_t>(meshGL.triVerts[3 * i + 1]),
185 static_cast<uint32_t>(meshGL.triVerts[3 * i + 2])));
186 }
187 }
188 }
189
190 CreateHalfedges(triVerts);
191 if (!IsManifold()) {
192 MarkFailure(Error::NotManifold);
193 return;
194 }
195
196 CalculateBBox();
197 if (!IsFinite()) {
198 MarkFailure(Error::NonFiniteVertex);
199 return;
200 }
201 SetPrecision(meshGL.precision);
202
203 SplitPinchedVerts();
204
205 CalculateNormals();
206
207 if (meshGL.runOriginalID.empty()) {
208 InitializeOriginal();
209 }
210
211 SimplifyTopology();
212 Finish();
213
214 // A Manifold created from an input mesh is never an original - the input is
215 // the original.
216 meshRelation_.originalID = -1;
217 }

219 inline void ForVert(int halfedge, std::function<void(int halfedge)> func) {
220 int current = halfedge;
221 do {
222 current = NextHalfedge(halfedge_[current].pairedHalfedge);
223 func(current);
224 } while (current != halfedge);
225 }
226
227 template <typename T>
228 void ForVert(
229 int halfedge, std::function<T(int halfedge)> transform,
230 std::function<void(int halfedge, const T& here, T& next)> binaryOp) {
231 T here = transform(halfedge);
232 int current = halfedge;
233 do {
234 const int nextHalfedge = NextHalfedge(halfedge_[current].pairedHalfedge);
235 T next = transform(nextHalfedge);
236 binaryOp(current, here, next);
237 here = next;
238 current = nextHalfedge;
239 } while (current != halfedge);
240 }
