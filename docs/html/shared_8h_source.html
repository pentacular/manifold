51inline mat3x2 GetAxisAlignedProjection(vec3 normal) {
52 vec3 absNormal = glm::abs(normal);
53 double xyzMax;
54 mat2x3 projection;
55 if (absNormal.z > absNormal.x && absNormal.z > absNormal.y) {
56 projection = mat2x3(1.0, 0.0, 0.0, //
57 0.0, 1.0, 0.0);
58 xyzMax = normal.z;
59 } else if (absNormal.y > absNormal.x) {
60 projection = mat2x3(0.0, 0.0, 1.0, //
61 1.0, 0.0, 0.0);
62 xyzMax = normal.y;
63 } else {
64 projection = mat2x3(0.0, 1.0, 0.0, //
65 0.0, 0.0, 1.0);
66 xyzMax = normal.x;
67 }
68 if (xyzMax < 0) projection[0] *= -1.0;
69 return glm::transpose(projection);
70}

72inline vec3 GetBarycentric(const vec3& v, const mat3& triPos,
73 double precision) {
74 const mat3 edges(triPos[2] - triPos[1], triPos[0] - triPos[2],
75 triPos[1] - triPos[0]);
76 const vec3 d2(glm::dot(edges[0], edges[0]), glm::dot(edges[1], edges[1]),
77 glm::dot(edges[2], edges[2]));
78 const int longSide = d2[0] > d2[1] && d2[0] > d2[2] ? 0
79 : d2[1] > d2[2] ? 1
80 : 2;
81 const vec3 crossP = glm::cross(edges[0], edges[1]);
82 const double area2 = glm::dot(crossP, crossP);
83 const double tol2 = precision * precision;
84
85 vec3 uvw(0);
86 for (const int i : {0, 1, 2}) {
87 const vec3 dv = v - triPos[i];
88 if (glm::dot(dv, dv) < tol2) {
89 // Return exactly equal if within tolerance of vert.
90 uvw[i] = 1;
91 return uvw;
92 }
93 }
94
95 if (d2[longSide] < tol2) { // point
96 return vec3(1, 0, 0);
97 } else if (area2 > d2[longSide] * tol2) { // triangle
98 for (const int i : {0, 1, 2}) {
99 const int j = Next3(i);
100 const vec3 crossPv = glm::cross(edges[i], v - triPos[j]);
101 const double area2v = glm::dot(crossPv, crossPv);
102 // Return exactly equal if within tolerance of edge.
103 uvw[i] = area2v < d2[i] * tol2 ? 0 : glm::dot(crossPv, crossP);
104 }
105 uvw /= (uvw[0] + uvw[1] + uvw[2]);
106 return uvw;
107 } else { // line
108 const int nextV = Next3(longSide);
109 const double alpha =
110 glm::dot(v - triPos[nextV], edges[longSide]) / d2[longSide];
111 uvw[longSide] = 0;
112 uvw[nextV] = 1 - alpha;
113 const int lastV = Next3(nextV);
114 uvw[lastV] = alpha;
115 return uvw;
116 }
117}

159struct TmpEdge {
160 int first, second, halfedgeIdx;
161
162 TmpEdge() {}
163 TmpEdge(int start, int end, int idx) {
164 first = std::min(start, end);
165 second = std::max(start, end);
166 halfedgeIdx = idx;
167 }
168
169 bool operator<(const TmpEdge& other) const {
170 return first == other.first ? second < other.second : first < other.first;
171 }
172};
175Vec<TmpEdge> inline CreateTmpEdges(const Vec<Halfedge>& halfedge) {
176 Vec<TmpEdge> edges(halfedge.size());
177 for_each_n(autoPolicy(edges.size()), countAt(0), edges.size(),
178 [&edges, &halfedge](const int idx) {
179 const Halfedge& half = halfedge[idx];
180 edges[idx] = TmpEdge(half.startVert, half.endVert,
181 half.IsForward() ? idx : -1);
182 });
183
184 size_t numEdge =
185 remove_if(edges.begin(), edges.end(),
186 [](const TmpEdge& edge) { return edge.halfedgeIdx < 0; }) -
187 edges.begin();
188 DEBUG_ASSERT(numEdge == halfedge.size() / 2, topologyErr, "Not oriented!");
189 edges.resize(numEdge);
190 return edges;
191}
