55inline double sind(double x) {
56 if (!std::isfinite(x)) return sin(x);
57 if (x < 0.0) return -sind(-x);
58 int quo;
59 x = remquo(fabs(x), 90.0, &quo);
60 switch (quo % 4) {
61 case 0:
62 return sin(glm::radians(x));
63 case 1:
64 return cos(glm::radians(x));
65 case 2:
66 return -sin(glm::radians(x));
67 case 3:
68 return -cos(glm::radians(x));
69 }
70 return 0.0;
71}

495 static void SetCircularSegments(int number) {
496 if (number < 3 && number != 0) return;
497 circularSegments_ = number;
498 }
499
507 static int GetCircularSegments(double radius) {
508 if (circularSegments_ > 0) return circularSegments_;
509 int nSegA = 360.0 / circularAngle_;
510 int nSegL = 2.0 * radius * glm::pi<double>() / circularEdgeLength_;
511 int nSeg = fmin(nSegA, nSegL) + 3;
512 nSeg -= nSeg % 4;
513 return std::max(nSeg, 3);
514 }

522 circularSegments_ = DEFAULT_SEGMENTS;
523 circularAngle_ = DEFAULT_ANGLE;
524 circularEdgeLength_ = DEFAULT_LENGTH;
